REM Title: HID Sentinel IR v1.0
REM Author: sucloudflare
REM Description: Lightweight defensive framework against BadUSB/HID attacks. Detects rogue USB devices and suspicious PowerShell activity in real-world scenarios.
REM Target: Windows 10/11
REM Version: 1.0
REM Category: Incident Response
REM =============================================

ATTACKMODE HID STORAGE
DEFAULT_DELAY 300
DELAY 2000

LED SETUP

REM === HID Sentinel - Active Protection ===
GUI r
DELAY 800
STRING powershell -NoP -Ep Bypass -W Hidden -C "
$Host.UI.RawUI.WindowTitle = 'HID Sentinel IR v1.0';
function Log { param($m,$l='INFO'); $t=Get-Date -Format 'HH:mm:ss'; \"[$t] [$l] $m\" | Out-File 'C:\Windows\Temp\sentinel.log' -Append -Encoding UTF8; Write-Host \"[$l] $m\" -ForegroundColor (@{INFO='White';WARN='Yellow';ALERT='Red';SUCCESS='Green'}[$l]) };
Log 'HID Sentinel IR v1.0 started - Active Protection' 'SUCCESS';

# Real-time Rogue USB Detection
Register-WmiEvent -Class Win32_DeviceChangeEvent -SourceIdentifier 'USBChange' -Action {
    if ($Event.SourceEventArgs.NewEvent.EventType -eq 2) {
        Start-Sleep -Milliseconds 700;
        $usb = Get-PnpDevice -Class USB | Where-Object Status -eq 'OK' | Select-Object -Last 5;
        foreach($d in $usb) {
            $name = $d.FriendlyName;
            if ($name -match '(?i)Rubber Ducky|BadUSB|HID Injector|Attack|Generic HID') {
                Log \"🚨 ROGUE BADUSB DEVICE DETECTED: $name\" 'ALERT';
            }
        }
    }
};

# Monitor suspicious PowerShell (common in real attacks)
while($true) {
    Get-Process -Name 'powershell*' -ErrorAction SilentlyContinue | Where-Object {
        $_.CommandLine -match '(?i)iex|downloadstring|invoke-webrequest|certutil|mshta|regsvr32|mimikatz|lazagne|credential|exfil'
    } | ForEach-Object {
        Log \"🚨 Suspicious PowerShell detected (PID: $($_.Id))\" 'ALERT';
    };
    Start-Sleep -Seconds 10;
}
"
ENTER

DELAY 1800
ALT y
DELAY 1200
ALT y
DELAY 3000

LED ATTACK

DELAY 3500
STRING notepad C:\Windows\Temp\sentinel.log
ENTER

LED FINISH
