REM Title: HID Sentinel IR v1.0
REM Author: sucloudflare
REM Description: Lightweight Anti-BadUSB defensive framework. Detects rogue USB devices via WMI and suspicious PowerShell activity with low noise.
REM Target: Windows 10/11
REM Version: 1.0
REM Category: Incident Response
REM =============================================

ATTACKMODE HID STORAGE
DEFAULT_DELAY 250
DELAY 1500

LED SETUP

REM === Launch HID Sentinel Defender ===
GUI r
DELAY 800
STRING powershell -NoP -Ep Bypass -W Hidden -C "$Host.UI.RawUI.WindowTitle='HID Sentinel IR v1.0'; function Log{param($m,$l='INFO');$t=Get-Date -Format 'HH:mm:ss';\"[$t] [$l] $m\"|Out-File 'C:\Windows\Temp\sentinel.log' -Append -Encoding UTF8;Write-Host \"[$l] $m\" -ForegroundColor(@{INFO='White';WARN='Yellow';ALERT='Red';SUCCESS='Green'}[$l])}; Log 'HID Sentinel IR v1.0 started - Anti-BadUSB Protection' 'SUCCESS'; Register-WmiEvent -Class Win32_DeviceChangeEvent -SourceIdentifier 'USBChange' -Action {if($Event.SourceEventArgs.NewEvent.EventType -eq 2){Start-Sleep -Milliseconds 700;$u=Get-PnpDevice -Class USB|? Status -eq 'OK'|Select -Last 5;foreach($d in $u){if($d.FriendlyName -match '(?i)Rubber Ducky|BadUSB|HID Injector|Attack'){Log \"ROGUE BADUSB DETECTED: $($d.FriendlyName)\" 'ALERT'}}}}; while($true){Get-Process -Name 'powershell*' -EA 0|?{$_.CommandLine -match '(?i)iex|downloadstring|invoke-webrequest'}|%{Log \"Suspicious PowerShell detected (PID: $($_.Id))\" 'WARN'};Start-Sleep -Seconds 12}"
ENTER

DELAY 2000
ALT y
DELAY 1200
ALT y
DELAY 3000

LED ATTACK

DELAY 4000
STRING notepad C:\Windows\Temp\sentinel.log
ENTER

LED FINISH
